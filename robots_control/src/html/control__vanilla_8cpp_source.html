125void takeoff_drone(int droneid,double z){
126
127 VehicleCommand msg{};
128 msg.param1 = 1;
129 msg.param2 = 6;
130 msg.command = VehicleCommand::VEHICLE_CMD_DO_SET_MODE; //set offboard mode
131 msg.target_system = NROVER+droneid+1;
132 msg.target_component = 0;
133 msg.source_system = 1;
134 msg.source_component = 1;
135 msg.from_external = true;
136
137 VehicleCommand msg2{};
138
139 msg2.param1 = 1.0; //1 = arm, 0 = disarm
140 //msg2.param2 = 0.0;
141 msg2.command = VehicleCommand::VEHICLE_CMD_COMPONENT_ARM_DISARM; //arm the drone
142 msg2.target_system = NROVER+droneid+1;
143 msg2.target_component = 0;
144 msg2.source_system = 1;
145 msg2.source_component = 1;
146 msg2.from_external = true;
147
148 OffboardControlMode msg3{};
149
150 msg3.position = true; //only control position
151 msg3.velocity = false;
152 msg3.acceleration = false;
153 msg3.attitude = false;
154 msg3.body_rate = false;
155
156 odom_callback_drone(droneid);
157 TrajectorySetpoint msg4{}; //set destination coordinates
158 msg4.x = x_drone[droneid];
159 msg4.y = y_drone[droneid];
160 msg4.z = -z; //the frame of reference has the z axis pointed "downward"
161 msg4.yaw = -PI;
162
163 int offboard_setpoint_counter_ = 0;
164 double dist_from_dest=DOUBLE_MAX;
165 double x_dest=0.0,y_dest=0.0,z_dest=-z;
166 int cont=0;
167
168 do{
169 odom_callback_drone(droneid); //get current position
170 dist_from_dest=sqrt(((x_drone[droneid]-x_dest)*(x_drone[droneid]-x_dest))+((y_drone[droneid]-y_dest)*(y_drone[droneid]-y_dest))+((z_drone[droneid]-z_dest)*(z_drone[droneid]-z_dest))); //calculate distance from target
171 #if DEBUG_MODE
172 if(droneid==0){
173 std::cout<<"distance from target:"<<dist_from_dest<<std::endl;
174 print_pos_drone(0);
175 }
176 #endif
177
178 if (offboard_setpoint_counter_ == 11) {
179 vehicle_command_publisher_drone[droneid]->publish(msg); //send offboard mode and arm commands
180 vehicle_command_publisher_drone[droneid]->publish(msg2);
181 }
182 offboard_control_mode_publisher_drone[droneid]->publish(msg3);
183 trajectory_setpoint_publisher_drone[droneid]->publish(msg4); //send some position command before the arm command to avoid arming and disarming without takeoff
184 if (offboard_setpoint_counter_ < 11) {
185 offboard_setpoint_counter_++;
186 }
187 rclcpp::spin_some(node_publisher_drone[droneid]);
188 delay(100);
189 cont++;
190 if(cont==100){ //print position every ~10 seconds
191 cont =0;
192 print_pos_drone(droneid);
193 }
194
195
196 }while(dist_from_dest>TARGET_PRECISION); //keep sen/ding command until it's close enough to destination
197 std::cout<<"Drone "<<droneid<<": takeoff sent\n";
198}

489void ui_callback(const interfaces::msg::Mymsg::SharedPtr msg){
490 cont_callback++;
491
492 int length = msg->length;
493
494 for(int i=0;i<length;i++){
495 int action = msg->action[i];
496 double x = msg->x[i];
497 double y = msg->y[i];
498 double z = msg->z[i];
499
500 if(msg->vehicle[i]=="d1"){ //each action is stored in the list of the vehicle that will perform that action (done in both first and second call)
501 drone_action[0].push_back(action);
502 drone_x[0].push_back(x);
503 drone_y[0].push_back(y);
504 drone_z[0].push_back(z);
505 }else if(msg->vehicle[i]=="d2"){
506 drone_action[1].push_back(action);
507 drone_x[1].push_back(x);
508 drone_y[1].push_back(y);
509 drone_z[1].push_back(z);
510 }else{
511 rover_action[0].push_back(action);
512 rover_x[0].push_back(x);
513 rover_y[0].push_back(y);
514 rover_z[0].push_back(z);
515 }
516
517 }
518
519 if(cont_callback==1){ //first time the callback is called
520 auto request = interfaces::msg::MsgRet();
521
522 //get the last position of each vehicle (origin if the vehicle didn't move) and save it to calculate the return path
523 if(rover_x[0].size()!=0){
524 request.x_rover = rover_x[0].back();
525 request.y_rover = rover_y[0].back();
526 }else{
527 request.x_rover = 0.0;
528 request.y_rover = 0.0;
529 }
530
531 if(drone_x[0].size()!=0){
532 request.x_drone_0 = drone_x[0].back();
533 request.y_drone_0 = drone_y[0].back();
534 request.z_drone_0 = drone_z[0].back();
535 }else{
536 request.x_drone_0 = 0.0;
537 request.y_drone_0 = 0.0;
538 request.z_drone_0 = 0.0;
539 }
540
541 if(drone_x[1].size()!=0){
542 request.x_drone_1 = drone_x[1].back();
543 request.y_drone_1 = drone_y[1].back();
544 request.z_drone_1 = drone_z[1].back();
545 }else{
546 request.x_drone_1 = 0.0;
547 request.y_drone_1 = 0.0;
548 request.z_drone_1 = 0.0;
549 }
550
551 delay(5000);
552 client_return->publish(request); //publish on the "/ret" topic the previous positions to calculate the return path
553 }
554
555 if(cont_callback==2){ //second time the callback is called (return path)
556 std::cout<<"Drone 0: # commands: "<<drone_action[0].size()<<"\nDrone 1: # commands: "<<drone_action[1].size()<<"\nRover: # commands: "<<rover_action[0].size()<<"\n";
557 //start all the missions

559 std::thread mission1(perform_mission_drone,1);
560 std::thread mission2(perform_mission_rover,0);
561
562 mission0.join();
563 mission1.join();
564 mission2.join();
565 }
